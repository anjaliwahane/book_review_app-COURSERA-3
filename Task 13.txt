const axios = require('axios');


// Function to search books by title

async function searchBooksByTitle(title) {

  try {

    const response = await axios.get('https://www.googleapis.com/books/v1/volumes', {

      params: {

        q: `intitle:${title}`, // Search by title

        maxResults: 10, // Limiting results to 10

      },

    });


    const books = response.data.items;

    if (books.length > 0) {

      console.log(`Books with title "${title}":`);

      books.forEach((book, index) => {

        console.log(`${index + 1}. ${book.volumeInfo.title}`);

      });

    } else {

      console.log(`No books found with title "${title}"`);

    }

  } catch (error) {

    console.error('Error fetching books:', error.message);

  }

}


// Usage: Pass the title to search for books

const bookTitle = 'Harry Potter'; // Replace with the desired book title

searchBooksByTitle(bookTitle);
